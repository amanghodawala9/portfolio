def stable_hash(val):
    return hashlib.sha256(str(val).encode()).hexdigest()[:16]

Data Transformation¶

In [8]:
df['created_at'] = pd.to_datetime(df['created_at'])
df['date'] = df['created_at'].dt.date
df['payload'] = df['payload'].apply(json.loads)
df['action'] = df['payload'].apply(lambda x: x['action'])
df['actor_id'] = df['payload'].apply(lambda x: x['actor_id'])

# assigning a unique number/hash to every user for privacy reasons
df['user_hash_id'] = df['actor_id'].astype(str).apply(stable_hash)
